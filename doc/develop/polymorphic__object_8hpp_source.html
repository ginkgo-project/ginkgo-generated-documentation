424 template <typename R, typename T>
425 std::unique_ptr<R, std::function<void(R*)>> copy_and_convert_to_impl(
426  std::shared_ptr<const Executor> exec, T* obj)
427 {
428  auto obj_as_r = dynamic_cast<R*>(obj);
429  if (obj_as_r != nullptr && obj->get_executor() == exec) {
430  return {obj_as_r, [](R*) {}};
431  } else {
432  auto copy = R::create(exec);
433  as<ConvertibleTo<std::decay_t<R>>>(obj)->convert_to(lend(copy));
434  return {copy.release(), std::default_delete<R>{}};
435  }
436 }
