154  std::vector<std::unique_ptr<unbatch_type>> unbatch() const
155  {
156  GKO_ASSERT_BATCH_HAS_SQUARE_MATRICES(this);
157  auto exec = this->get_executor();
158  auto unbatch_mats = std::vector<std::unique_ptr<unbatch_type>>{};
159  const auto c_entry_size =
160  std::min(this->get_size().at(0)[0], this->get_size().at(0)[1]);
161  for (size_type b = 0; b < this->get_num_batch_entries(); ++b) {
162  auto mat = unbatch_type::create(exec, this->get_size().at(b)[0]);
163  if (this->get_size().stores_equal_sizes()) {
164  exec->copy_from(exec.get(), mat->get_size()[0],
165  this->get_const_values() + b * c_entry_size,
166  mat->get_values());
167  } else {
168  exec->copy_from(
169  exec.get(), mat->get_size()[0],
170  this->get_const_values() +
171  num_elems_per_batch_cumul_.get_const_data()[b],
172  mat->get_values());
173  }
174  unbatch_mats.emplace_back(std::move(mat));
175  }
176  return unbatch_mats;
177  }

298  size_type compute_batch_mem(const batch_dim<2>& sizes) const
299  {
300  if (sizes.stores_equal_sizes()) {
301  const size_type n_values_per_entry =
302  std::min(sizes.at(0)[0], sizes.at(0)[1]);
303  return n_values_per_entry * sizes.get_num_batch_entries();
304  }
305  size_type mem_req = 0;
306  for (auto i = 0; i < sizes.get_num_batch_entries(); ++i) {
307  mem_req += std::min(sizes.at(i)[0], sizes.at(i)[1]);
308  }
309  return mem_req;

329  inline Array<size_type> compute_num_elems_per_batch_cumul(
330  std::shared_ptr<const Executor> exec, const batch_dim<2>& sizes)
331  {
332  auto num_elems = Array<size_type>(exec->get_master(),
333  sizes.get_num_batch_entries() + 1);
334  num_elems.get_data()[0] = 0;
335  for (auto i = 0; i < sizes.get_num_batch_entries(); ++i) {
336  num_elems.get_data()[i + 1] =
337  num_elems.get_data()[i] +
338  std::min(sizes.at(i)[0], sizes.at(i)[1]);
339  }
340  num_elems.set_executor(exec);
341  return num_elems;
342  }

420  BatchDiagonal(std::shared_ptr<const Executor> exec,
421  const size_type num_duplications,
422  const BatchDiagonal<value_type>* const input)
423  : EnableBatchLinOp<BatchDiagonal>(
424  exec, gko::batch_dim<2>(
425  input->get_num_batch_entries() * num_duplications,
426  input->get_size().at(0)))
427  {
428  const auto in_batch_entries = input->get_num_batch_entries();
429  const bool non_uniform = !input->get_size().stores_equal_sizes();
430  std::vector<dim<2>> nu_dim;
431  if (non_uniform) {
432  nu_dim.resize(input->get_num_batch_entries() * num_duplications);
433  for (size_type i = 0; i < num_duplications; ++i) {
434  for (size_type batch_idx = i * in_batch_entries;
435  batch_idx < (i + 1) * in_batch_entries; batch_idx++) {
436  nu_dim[batch_idx] =
437  input->get_size().at(batch_idx - i * in_batch_entries);
438  }
439  }
440  this->set_size(batch_dim<2>(nu_dim));
441  num_elems_per_batch_cumul_ = compute_num_elems_per_batch_cumul(
442  exec->get_master(), this->get_size());
443  }
444  values_ = Array<ValueType>(exec, compute_batch_mem(this->get_size()));
445  size_type offset = 0;
446  for (size_type i = 0; i < num_duplications; ++i) {
447  if (non_uniform) {
448  for (size_type batch_idx = i * in_batch_entries;
449  batch_idx < (i + 1) * in_batch_entries; batch_idx++) {
450  nu_dim[batch_idx] =
451  input->get_size().at(batch_idx - i * in_batch_entries);
452  }
453  }
454  exec->copy_from(
455  input->get_executor().get(), input->get_num_stored_elements(),
456  input->get_const_values(), this->get_values() + offset);
457  offset += input->get_num_stored_elements();
458  }
459  }

554 template <typename ValueType>
555 std::unique_ptr<matrix::BatchDiagonal<ValueType>> batch_diagonal_initialize(
556  std::initializer_list<std::initializer_list<ValueType>> vals,
557  std::shared_ptr<const Executor> exec)
558 {
559  using batch_diag = matrix::BatchDiagonal<ValueType>;
560  size_type num_batch_entries = vals.size();
561  std::vector<size_type> num_rows(num_batch_entries);
562  std::vector<dim<2>> sizes(num_batch_entries);
563  auto vals_begin = begin(vals);
564  for (size_type b = 0; b < num_batch_entries; ++b) {
565  num_rows[b] = vals_begin->size();
566  sizes[b] = dim<2>(num_rows[b], num_rows[b]);
567  vals_begin++;
568  }
569  auto b_size = batch_dim<2>(sizes);
570  auto tmp = batch_diag::create(exec->get_master(), b_size);
571  size_type batch = 0;
572  for (const auto& b : vals) {
573  size_type idx = 0;
574  for (const auto& elem : b) {
575  tmp->at(batch, idx) = elem;
576  ++idx;
577  }
578  ++batch;
579  }
580  auto mtx = batch_diag::create(exec);
581  tmp->move_to(mtx.get());
582  return mtx;
583 }
