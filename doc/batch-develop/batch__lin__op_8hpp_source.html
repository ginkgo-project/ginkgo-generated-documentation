116  return x.num_batch_entries_ == y.num_batch_entries_ &&
117  x.common_stride_ == y.common_stride_;
118  } else {

120  }

158 private:
159  inline void check_equal_strides()
160  {
161  for (size_type b = 1; b < num_batch_entries_; ++b) {
162  if (strides_[0] != strides_[b]) {
163  equal_strides_ = false;
164  common_stride_ = 0;
165  return;
166  }
167  }
168  equal_strides_ = true;
169  common_stride_ = strides_[0];
170  }

686  void batch_scale(const BatchLinOp* const left_scale,
687  const BatchLinOp* const right_scale) override
688  {
689  if (!left_scale && !right_scale) {
690  return;
691  }
692 
693  /* TODO: Somehow restrict BatchScalable and EnableBatchScaling to
694  * BatchLinOp so that the following dynamic_cast is not needed.
695  */
696  if (auto blp = dynamic_cast<BatchLinOp*>(this)) {
697  GKO_ASSERT_BATCH_CONFORMANT(left_scale, blp);
698  GKO_ASSERT_BATCH_REVERSE_CONFORMANT(right_scale, blp);
699  } else {
700  GKO_NOT_SUPPORTED(this);
701  }
702 
703  left_scale_ = left_scale;
704  right_scale_ = right_scale;
705 
706  this->batch_scale_impl(left_scale, right_scale);
707  }
