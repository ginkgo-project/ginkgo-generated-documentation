96  explicit ResidualNormBase(std::shared_ptr<const gko::Executor> exec,
97  const CriterionArgs& args,
98  remove_complex<ValueType> reduction_factor,
99  mode baseline)
100  : EnablePolymorphicObject<ResidualNormBase, Criterion>(exec),
101  reduction_factor_{reduction_factor},
102  device_storage_{exec, 2},
103  baseline_{baseline},
104  system_matrix_{args.system_matrix},
105  b_{args.b},
106  one_{gko::initialize<Vector>({1}, exec)},
107  neg_one_{gko::initialize<Vector>({-1}, exec)}
108  {
109  switch (baseline_) {
110  case mode::initial_resnorm: {
111  if (args.initial_residual == nullptr) {
112  if (args.system_matrix == nullptr || args.b == nullptr ||
113  args.x == nullptr) {
114  GKO_NOT_SUPPORTED(nullptr);
115  } else {
116  this->starting_tau_ = NormVector::create(
117  exec, dim<2>{1, args.b->get_size()[1]});
118  auto b_clone = share(args.b->clone());
119  args.system_matrix->apply(neg_one_.get(), args.x,
120  one_.get(), b_clone.get());
121  if (auto vec =
122  std::dynamic_pointer_cast<const ComplexVector>(
123  b_clone)) {
124  vec->compute_norm2(this->starting_tau_.get());
125  } else if (auto vec =
126  std::dynamic_pointer_cast<const Vector>(
127  b_clone)) {
128  vec->compute_norm2(this->starting_tau_.get());
129  } else {
130  GKO_NOT_SUPPORTED(nullptr);
131  }
132  }
133  } else {
134  this->starting_tau_ = NormVector::create(
135  exec, dim<2>{1, args.initial_residual->get_size()[1]});
136  if (dynamic_cast<const ComplexVector*>(args.initial_residual)) {
137  auto dense_r = as<ComplexVector>(args.initial_residual);
138  dense_r->compute_norm2(this->starting_tau_.get());
139  } else {
140  auto dense_r = as<Vector>(args.initial_residual);
141  dense_r->compute_norm2(this->starting_tau_.get());
142  }
143  }
144  break;
145  }
146  case mode::rhs_norm: {
147  if (args.b == nullptr) {
148  GKO_NOT_SUPPORTED(nullptr);
149  }
150  this->starting_tau_ =
151  NormVector::create(exec, dim<2>{1, args.b->get_size()[1]});
152  if (dynamic_cast<const ComplexVector*>(args.b.get())) {
153  auto dense_rhs = as<ComplexVector>(args.b);
154  dense_rhs->compute_norm2(this->starting_tau_.get());
155  } else {
156  auto dense_rhs = as<Vector>(args.b);
157  dense_rhs->compute_norm2(this->starting_tau_.get());
158  }
159  break;
160  }
161  case mode::absolute: {
162  if (args.b == nullptr) {
163  GKO_NOT_SUPPORTED(nullptr);
164  }
165  this->starting_tau_ =
166  NormVector::create(exec, dim<2>{1, args.b->get_size()[1]});
167  this->starting_tau_->fill(gko::one<remove_complex<ValueType>>());
168  break;
169  }
170  default:
171  GKO_NOT_SUPPORTED(nullptr);
172  }
173  this->u_dense_tau_ =
174  NormVector::create_with_config_of(this->starting_tau_.get());
175  }
176 
177  remove_complex<ValueType> reduction_factor_{};
178  std::unique_ptr<NormVector> starting_tau_{};
179  std::unique_ptr<NormVector> u_dense_tau_{};
180  /* Contains device side: all_converged and one_changed booleans */
181  array<bool> device_storage_;
182 
183 private:
184  mode baseline_{mode::rhs_norm};
185  std::shared_ptr<const LinOp> system_matrix_{};
186  std::shared_ptr<const LinOp> b_{};
187  /* one/neg_one for residual computation */
188  std::shared_ptr<const Vector> one_{};
189  std::shared_ptr<const Vector> neg_one_{};
190 };
