457 template <typename Matrix, typename... TArgs>
458 std::unique_ptr<Matrix> initialize(
459  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
460  vals,
461  std::shared_ptr<const Executor> exec, TArgs&&... create_args)
462 {
463  using batch_multi_vector = MultiVector<typename Matrix::value_type>;
464  size_type num_batch_items = vals.size();
465  GKO_THROW_IF_INVALID(num_batch_items > 0, "Input data is empty");
466  auto vals_begin = begin(vals);
467  size_type common_num_rows = vals_begin ? vals_begin->size() : 0;
468  auto common_size = dim<2>(common_num_rows, 1);
469  for (auto& val : vals) {
470  GKO_ASSERT_EQ(common_num_rows, val.size());
471  }
472  auto b_size = batch_dim<2>(num_batch_items, common_size);
473  auto tmp = batch_multi_vector::create(exec->get_master(), b_size);
474  size_type batch = 0;
475  for (const auto& b : vals) {
476  size_type idx = 0;
477  for (const auto& elem : b) {
478  tmp->at(batch, idx) = elem;
479  ++idx;
480  }
481  ++batch;
482  }
483  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
484  tmp->move_to(mtx);
485  return mtx;
486 }
