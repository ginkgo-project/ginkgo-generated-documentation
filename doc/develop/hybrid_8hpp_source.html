234  size_type compute_ell_num_stored_elements_per_row(
235  array<size_type>* row_nnz) const override
236  {
237  auto row_nnz_val = row_nnz->get_data();
238  auto num_rows = row_nnz->get_num_elems();
239  if (num_rows == 0) {
240  return 0;
241  }
242  std::sort(row_nnz_val, row_nnz_val + num_rows);
243  if (percent_ < 1) {
244  auto percent_pos = static_cast<size_type>(num_rows * percent_);
245  return row_nnz_val[percent_pos];
246  } else {
247  return row_nnz_val[num_rows - 1];
248  }
249  }

752 template <typename ValueType, typename IndexType>
753 template <typename HybType>
754 std::shared_ptr<typename HybType::strategy_type>
755 Hybrid<ValueType, IndexType>::get_strategy() const
756 {
757  static_assert(
758  std::is_same<HybType, Hybrid<typename HybType::value_type,
759  typename HybType::index_type>>::value,
760  "The given `HybType` type must be of type `matrix::Hybrid`!");
761 
762  std::shared_ptr<typename HybType::strategy_type> strategy;
763  if (std::dynamic_pointer_cast<automatic>(strategy_)) {
764  strategy = std::make_shared<typename HybType::automatic>();
765  } else if (auto temp = std::dynamic_pointer_cast<minimal_storage_limit>(
766  strategy_)) {
767  // minimal_storage_limit is related to ValueType and IndexType size.
768  if (sizeof(value_type) == sizeof(typename HybType::value_type) &&
769  sizeof(index_type) == sizeof(typename HybType::index_type)) {
770  strategy =
771  std::make_shared<typename HybType::minimal_storage_limit>();
772  } else {
773  strategy = std::make_shared<typename HybType::imbalance_limit>(
774  temp->get_percentage());
775  }
776  } else if (auto temp = std::dynamic_pointer_cast<imbalance_bounded_limit>(
777  strategy_)) {
778  strategy = std::make_shared<typename HybType::imbalance_bounded_limit>(
779  temp->get_percentage(), temp->get_ratio());
780  } else if (auto temp =
781  std::dynamic_pointer_cast<imbalance_limit>(strategy_)) {
782  strategy = std::make_shared<typename HybType::imbalance_limit>(
783  temp->get_percentage());
784  } else if (auto temp = std::dynamic_pointer_cast<column_limit>(strategy_)) {
785  strategy = std::make_shared<typename HybType::column_limit>(
786  temp->get_num_columns());
787  } else {
788  GKO_NOT_SUPPORTED(strategy_);
789  }
790  return strategy;
791 }
