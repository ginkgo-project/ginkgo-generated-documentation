154  std::vector<std::unique_ptr<unbatch_type>> unbatch() const
155  {
156  auto exec = this->get_executor();
157  auto unbatch_mats = std::vector<std::unique_ptr<unbatch_type>>{};
158  size_type num_nnz =
159  this->get_num_stored_elements() / this->get_num_batch_entries();
160  size_type offset = 0;
161  for (size_type b = 0; b < this->get_num_batch_entries(); ++b) {
162  auto mat = unbatch_type::create(
163  exec, this->get_size().at(b),
164  this->get_num_stored_elements_per_row().at(b),
165  this->get_stride().at(b));
166  exec->copy_from(exec.get(), num_nnz,
167  this->get_const_values() + offset,
168  mat->get_values());
169  exec->copy_from(exec.get(), num_nnz, this->get_const_col_idxs(),
170  mat->get_col_idxs());
171  unbatch_mats.emplace_back(std::move(mat));
172  offset += num_nnz;
173  }
174  return unbatch_mats;
175  }

481  template <typename MatrixType>
482  void batch_duplicator(std::shared_ptr<const Executor> exec,
483  const size_type num_batch_entries,
484  const size_type col_idxs_size,
485  const MatrixType* input,
486  BatchEll<value_type, index_type>* output)
487  {
488  auto col_idxs = output->get_col_idxs();
489  auto values = output->get_values();
490  size_type offset = 0;
491  size_type num_nnz = input->get_num_stored_elements();
492  for (size_type i = 0; i < num_batch_entries; ++i) {
493  exec->copy_from(input->get_executor().get(), num_nnz,
494  input->get_const_values(), values + offset);
495  offset += num_nnz;
496  }
497  exec->copy_from(input->get_executor().get(), col_idxs_size,
498  input->get_const_col_idxs(), col_idxs);
499  }

507  size_type linearize_index(size_type batch_id, size_type row,
508  size_type col) const noexcept
509  {
510  size_type num_elems_per_batch =
511  values_.get_num_elems() / this->get_num_batch_entries();
512  return batch_id * num_elems_per_batch + row +
513  stride_.at(batch_id) * col;

516  batch_stride get_stride_from_dim(const batch_dim<2>& dim)
517  {
518  if (dim.stores_equal_sizes()) {
519  return batch_stride(dim.get_num_batch_entries(), dim.at(0)[0]);
520  } else {
521  std::vector<size_type> strides;
522  strides.reserve(dim.get_num_batch_entries());
523  auto batch_sizes = dim.get_batch_sizes();
524  for (auto dims : batch_sizes) {
525  strides.push_back(dims[0]);
526  }
527  return batch_stride(strides);
528  }
