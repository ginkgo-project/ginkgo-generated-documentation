749  template <typename T>
750  void copy_from(const Executor* src_exec, size_type num_elems,
751  const T* src_ptr, T* dest_ptr) const
752  {
753  const auto src_loc = reinterpret_cast<uintptr>(src_ptr);
754  const auto dest_loc = reinterpret_cast<uintptr>(dest_ptr);
755  this->template log<log::Logger::copy_started>(
756  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
757  if (this != src_exec) {
758  src_exec->template log<log::Logger::copy_started>(
759  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
760  }
761  try {
762  this->raw_copy_from(src_exec, num_elems * sizeof(T), src_ptr,
763  dest_ptr);
764  } catch (NotSupported&) {
765 #if (GKO_VERBOSE_LEVEL >= 1) && !defined(NDEBUG)
766  // Unoptimized copy. Try to go through the masters.
767  // output to log when verbose >= 1 and debug build
768  std::clog << "Not direct copy. Try to copy data from the masters."
769  << std::endl;
770 #endif
771  auto src_master = src_exec->get_master().get();
772  if (num_elems > 0 && src_master != src_exec) {
773  auto* master_ptr = src_exec->get_master()->alloc<T>(num_elems);
774  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
775  master_ptr);
776  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
777  src_master->free(master_ptr);
778  }
779  }
780  this->template log<log::Logger::copy_completed>(
781  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
782  if (this != src_exec) {
783  src_exec->template log<log::Logger::copy_completed>(
784  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
785  }
786  }
