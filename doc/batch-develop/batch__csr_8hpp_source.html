132  std::vector<std::unique_ptr<unbatch_type>> unbatch() const
133  {
134  auto exec = this->get_executor();
135  auto unbatch_mats = std::vector<std::unique_ptr<unbatch_type>>{};
136  size_type num_nnz =
137  this->get_num_stored_elements() / this->get_num_batch_entries();
138  size_type offset = 0;
139  for (size_type b = 0; b < this->get_num_batch_entries(); ++b) {
140  auto mat =
141  unbatch_type::create(exec, this->get_size().at(b), num_nnz);
142  exec->copy_from(exec.get(), num_nnz,
143  this->get_const_values() + offset,
144  mat->get_values());
145  exec->copy_from(exec.get(), num_nnz, this->get_const_col_idxs(),
146  mat->get_col_idxs());
147  exec->copy_from(exec.get(), this->get_size().at(b)[0] + 1,
148  this->get_const_row_ptrs(), mat->get_row_ptrs());
149  unbatch_mats.emplace_back(std::move(mat));
150  offset += num_nnz;
151  }
152  return unbatch_mats;
153  }

411  void batch_duplicator(std::shared_ptr<const Executor> exec,
412  const size_type num_batch_entries,
413  const size_type col_idxs_size,
414  const MatrixType* input,
415  BatchCsr<value_type, index_type>* output)
416  {

423  exec->copy_from(input->get_executor().get(), num_nnz,
