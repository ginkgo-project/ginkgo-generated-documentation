257 class precision_reduction {
258 public:
262  using storage_type = uint8;
263 
264 private:
265  static constexpr auto nonpreserving_bits = 4u;
266  static constexpr auto preserving_bits =
267  byte_size * sizeof(storage_type) - nonpreserving_bits;
268  static constexpr auto nonpreserving_mask =
269  storage_type{(0x1 << nonpreserving_bits) - 1};
270  static constexpr auto preserving_mask =
271  storage_type{(0x1 << preserving_bits) - 1} << nonpreserving_bits;
272 
273 public:
279  GKO_ATTRIBUTES constexpr precision_reduction() noexcept : data_{0x0} {}
280 
288  GKO_ATTRIBUTES constexpr precision_reduction(
289  storage_type preserving, storage_type nonpreserving) noexcept
290  : data_((GKO_ASSERT(preserving < (0x1 << preserving_bits) - 1),
291  GKO_ASSERT(nonpreserving < (0x1 << nonpreserving_bits) - 1),
292  (preserving << nonpreserving_bits) | nonpreserving))
293  {}
294 
300  GKO_ATTRIBUTES constexpr operator storage_type() const noexcept
301  {
302  return data_;
303  }
304 
310  GKO_ATTRIBUTES constexpr storage_type get_preserving() const noexcept
311  {
312  return (data_ & preserving_mask) >> nonpreserving_bits;
313  }
314 
320  GKO_ATTRIBUTES constexpr storage_type get_nonpreserving() const noexcept
321  {
322  return data_ & nonpreserving_mask;
323  }

332  GKO_ATTRIBUTES constexpr static precision_reduction autodetect() noexcept
333  {
334  return precision_reduction{preserving_mask | nonpreserving_mask};
335  }

348  GKO_ATTRIBUTES constexpr static precision_reduction common(
349  precision_reduction x, precision_reduction y) noexcept
350  {
351  return precision_reduction(
352  min(x.data_ & preserving_mask, y.data_ & preserving_mask) |
353  min(x.data_ & nonpreserving_mask, y.data_ & nonpreserving_mask));
354  }

356 private:
357  GKO_ATTRIBUTES constexpr precision_reduction(storage_type data)
358  : data_{data}

362  storage_type y) noexcept
363  {
