127 template <typename ValueType, typename Function>
128 void precision_dispatch_real_complex(Function fn, const LinOp* in, LinOp* out)
129 {
130  // do we need to convert complex Dense to real Dense?
131  // all real dense vectors are intra-convertible, thus by casting to
132  // ConvertibleTo<matrix::Dense<>>, we can check whether a LinOp is a real
133  // dense matrix:
134  auto complex_to_real =
135  !(is_complex<ValueType>() ||
136  dynamic_cast<const ConvertibleTo<matrix::Dense<>>*>(in));
137  if (complex_to_real) {
138  auto dense_in = make_temporary_conversion<to_complex<ValueType>>(in);
139  auto dense_out = make_temporary_conversion<to_complex<ValueType>>(out);
140  using Dense = matrix::Dense<ValueType>;
141  // These dynamic_casts are only needed to make the code compile
142  // If ValueType is complex, this branch will never be taken
143  // If ValueType is real, the cast is a no-op
144  fn(dynamic_cast<const Dense*>(dense_in->create_real_view().get()),
145  dynamic_cast<Dense*>(dense_out->create_real_view().get()));
146  } else {
147  precision_dispatch<ValueType>(fn, in, out);
148  }
149 }

258 template <typename ValueType, typename Function>
259 void mixed_precision_dispatch(Function fn, const LinOp* in, LinOp* out)
260 {
261 #ifdef GINKGO_MIXED_PRECISION
262  using fst_type = matrix::Dense<ValueType>;
263  using snd_type = matrix::Dense<next_precision<ValueType>>;
264  if (auto dense_in = dynamic_cast<const fst_type*>(in)) {
265  if (auto dense_out = dynamic_cast<fst_type*>(out)) {
266  fn(dense_in, dense_out);
267  } else if (auto dense_out = dynamic_cast<snd_type*>(out)) {
268  fn(dense_in, dense_out);
269  } else {
270  GKO_NOT_SUPPORTED(out);
271  }
272  } else if (auto dense_in = dynamic_cast<const snd_type*>(in)) {
273  if (auto dense_out = dynamic_cast<fst_type*>(out)) {
274  fn(dense_in, dense_out);
275  } else if (auto dense_out = dynamic_cast<snd_type*>(out)) {
276  fn(dense_in, dense_out);
277  } else {
278  GKO_NOT_SUPPORTED(out);
279  }
280  } else {
281  GKO_NOT_SUPPORTED(in);
282  }
283 #else
284  precision_dispatch<ValueType>(fn, in, out);
285 #endif
286 }
