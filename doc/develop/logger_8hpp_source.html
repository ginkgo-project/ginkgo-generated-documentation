765 template <typename ConcreteLoggable, typename PolymorphicBase = Loggable>
766 class EnableLogging : public PolymorphicBase {
767 public:
768  void add_logger(std::shared_ptr<const Logger> logger) override
769  {
770  loggers_.push_back(logger);
771  }
772 
773  void remove_logger(const Logger* logger) override
774  {
775  auto idx =
776  find_if(begin(loggers_), end(loggers_),
777  [&logger](const auto& l) { return l.get() == logger; });
778  if (idx != end(loggers_)) {
779  loggers_.erase(idx);
780  } else {
781  throw OutOfBoundsError(__FILE__, __LINE__, loggers_.size(),
782  loggers_.size());
783  }
784  }

807  template <size_type Event, typename ConcreteLoggableT, typename = void>
808  struct propagate_log_helper {
809  template <typename... Args>
810  static void propagate_log(const ConcreteLoggableT*, Args&&...)
811  {}
812  };
813 
814  template <size_type Event, typename ConcreteLoggableT>
815  struct propagate_log_helper<
816  Event, ConcreteLoggableT,
817  xstd::void_t<
818  decltype(std::declval<ConcreteLoggableT>().get_executor())>> {
819  template <typename... Args>
820  static void propagate_log(const ConcreteLoggableT* loggable,
821  Args&&... args)
822  {
823  const auto exec = loggable->get_executor();
824  if (exec->should_propagate_log()) {
825  for (auto& logger : exec->get_loggers()) {
826  if (logger->needs_propagation()) {
827  logger->template on<Event>(std::forward<Args>(args)...);
828  }
829  }
830  }
831  }
832  };
833 
834 protected:
835  template <size_type Event, typename... Params>
836  void log(Params&&... params) const
837  {
838  propagate_log_helper<Event, ConcreteLoggable>::propagate_log(
839  static_cast<const ConcreteLoggable*>(this),
840  std::forward<Params>(params)...);
841  for (auto& logger : loggers_) {
842  logger->template on<Event>(std::forward<Params>(params)...);
843  }
844  }
