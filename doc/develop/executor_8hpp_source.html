699  template <typename T>
700  void copy_from(const Executor* src_exec, size_type num_elems,
701  const T* src_ptr, T* dest_ptr) const
702  {
703  const auto src_loc = reinterpret_cast<uintptr>(src_ptr);
704  const auto dest_loc = reinterpret_cast<uintptr>(dest_ptr);
705  this->template log<log::Logger::copy_started>(
706  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
707  if (this != src_exec) {
708  src_exec->template log<log::Logger::copy_started>(
709  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
710  }
711  try {
712  this->raw_copy_from(src_exec, num_elems * sizeof(T), src_ptr,
713  dest_ptr);
714  } catch (NotSupported&) {
715 #if (GKO_VERBOSE_LEVEL >= 1) && !defined(NDEBUG)
716  // Unoptimized copy. Try to go through the masters.
717  // output to log when verbose >= 1 and debug build
718  std::clog << "Not direct copy. Try to copy data from the masters."
719  << std::endl;
720 #endif
721  auto src_master = src_exec->get_master().get();
722  if (num_elems > 0 && src_master != src_exec) {
723  auto* master_ptr = src_exec->get_master()->alloc<T>(num_elems);
724  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
725  master_ptr);
726  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
727  src_master->free(master_ptr);
728  }
729  }
730  this->template log<log::Logger::copy_completed>(
731  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
732  if (this != src_exec) {
733  src_exec->template log<log::Logger::copy_completed>(
734  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
735  }
736  }
