692  template <size_type Event, typename ConcreteLoggableT>
693  struct propagate_log_helper<
694  Event, ConcreteLoggableT,
695  xstd::void_t<
696  decltype(std::declval<ConcreteLoggableT>().get_executor())>> {
697  template <typename... Args>
698  static void propagate_log(const ConcreteLoggableT* loggable,
699  Args&&... args)
700  {
701  const auto exec = loggable->get_executor();
702  if (exec->should_propagate_log()) {
703  for (auto& logger : exec->get_loggers()) {
704  if (logger->needs_propagation()) {
705  logger->template on<Event>(std::forward<Args>(args)...);
706  }
707  }
708  }
709  }
710  };

712 protected:

714  void log(Params&&... params) const

717  static_cast<const ConcreteLoggable*>(this),
718  std::forward<Params>(params)...);
