129 template <typename ValueType, typename Function>
130 void precision_dispatch_real_complex(Function fn, const LinOp* in, LinOp* out)
131 {
132  // do we need to convert complex Dense to real Dense?
133  // all real dense vectors are intra-convertible, thus by casting to
134  // ConvertibleTo<matrix::Dense<>>, we can check whether a LinOp is a real
135  // dense matrix:
136  auto complex_to_real =
137  !(is_complex<ValueType>() ||
138  dynamic_cast<const ConvertibleTo<matrix::Dense<>>*>(in));
139  if (complex_to_real) {
140  auto dense_in = make_temporary_conversion<to_complex<ValueType>>(in);
141  auto dense_out = make_temporary_conversion<to_complex<ValueType>>(out);
142  using Dense = matrix::Dense<ValueType>;
143  // These dynamic_casts are only needed to make the code compile
144  // If ValueType is complex, this branch will never be taken
145  // If ValueType is real, the cast is a no-op
146  fn(dynamic_cast<const Dense*>(dense_in->create_real_view().get()),
147  dynamic_cast<Dense*>(dense_out->create_real_view().get()));
148  } else {
149  precision_dispatch<ValueType>(fn, in, out);
150  }
151 }

260 template <typename ValueType, typename Function>
261 void mixed_precision_dispatch(Function fn, const LinOp* in, LinOp* out)
262 {
263 #ifdef GINKGO_MIXED_PRECISION
264  using fst_type = matrix::Dense<ValueType>;
265  using snd_type = matrix::Dense<next_precision<ValueType>>;
266  if (auto dense_in = dynamic_cast<const fst_type*>(in)) {
267  if (auto dense_out = dynamic_cast<fst_type*>(out)) {
268  fn(dense_in, dense_out);
269  } else if (auto dense_out = dynamic_cast<snd_type*>(out)) {
270  fn(dense_in, dense_out);
271  } else {
272  GKO_NOT_SUPPORTED(out);
273  }
274  } else if (auto dense_in = dynamic_cast<const snd_type*>(in)) {
275  if (auto dense_out = dynamic_cast<fst_type*>(out)) {
276  fn(dense_in, dense_out);
277  } else if (auto dense_out = dynamic_cast<snd_type*>(out)) {
278  fn(dense_in, dense_out);
279  } else {
280  GKO_NOT_SUPPORTED(out);
281  }
282  } else {
283  GKO_NOT_SUPPORTED(in);
284  }
285 #else
286  precision_dispatch<ValueType>(fn, in, out);
287 #endif
288 }
