676 template <typename ConcreteLoggable, typename PolymorphicBase = Loggable>
677 class EnableLogging : public PolymorphicBase {
678 public:
679  void add_logger(std::shared_ptr<const Logger> logger) override
680  {
681  loggers_.push_back(logger);
682  }
683 
684  void remove_logger(const Logger* logger) override
685  {
686  auto idx =
687  find_if(begin(loggers_), end(loggers_),
688  [&logger](const auto& l) { return lend(l) == logger; });
689  if (idx != end(loggers_)) {
690  loggers_.erase(idx);
691  } else {
692  throw OutOfBoundsError(__FILE__, __LINE__, loggers_.size(),
693  loggers_.size());
694  }
695  }
696 
697  const std::vector<std::shared_ptr<const Logger>>& get_loggers()
698  const override
699  {
700  return loggers_;
701  }
702 
703  void clear_loggers() override { loggers_.clear(); }
704 
705 protected:
706  template <size_type Event, typename... Params>
707  void log(Params&&... params) const
708  {
709  for (auto& logger : loggers_) {
710  logger->template on<Event>(std::forward<Params>(params)...);
711  }
712  }
