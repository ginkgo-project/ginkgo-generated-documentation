672  template <typename T>
673  void copy_from(const Executor* src_exec, size_type num_elems,
674  const T* src_ptr, T* dest_ptr) const
675  {
676  const auto src_loc = reinterpret_cast<uintptr>(src_ptr);
677  const auto dest_loc = reinterpret_cast<uintptr>(dest_ptr);
678  this->template log<log::Logger::copy_started>(
679  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
680  if (this != src_exec) {
681  src_exec->template log<log::Logger::copy_started>(
682  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
683  }
684  try {
685  this->raw_copy_from(src_exec, num_elems * sizeof(T), src_ptr,
686  dest_ptr);
687  } catch (NotSupported&) {
688 #if (GKO_VERBOSE_LEVEL >= 1) && !defined(NDEBUG)
689  // Unoptimized copy. Try to go through the masters.
690  // output to log when verbose >= 1 and debug build
691  std::clog << "Not direct copy. Try to copy data from the masters."
692  << std::endl;
693 #endif
694  auto src_master = src_exec->get_master().get();
695  if (num_elems > 0 && src_master != src_exec) {
696  auto* master_ptr = src_exec->get_master()->alloc<T>(num_elems);
697  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
698  master_ptr);
699  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
700  src_master->free(master_ptr);
701  }
702  }
703  this->template log<log::Logger::copy_completed>(
704  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
705  if (this != src_exec) {
706  src_exec->template log<log::Logger::copy_completed>(
707  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
708  }
709  }
