501 template <typename R, typename T>
502 std::unique_ptr<R, std::function<void(R*)>> copy_and_convert_to_impl(
503  std::shared_ptr<const Executor> exec, T* obj)
504 {
505  auto obj_as_r = dynamic_cast<R*>(obj);
506  if (obj_as_r != nullptr && obj->get_executor() == exec) {
507  return {obj_as_r, [](R*) {}};
508  } else {
509  auto copy = R::create(exec);
510  as<ConvertibleTo<std::decay_t<R>>>(obj)->convert_to(lend(copy));
511  return {copy.release(), std::default_delete<R>{}};
512  }
513 }
