598  void resize_and_reset(size_type num_elems)
599  {
600  if (num_elems == num_elems_) {
601  return;
602  }
603  if (exec_ == nullptr) {
604  throw gko::NotSupported(__FILE__, __LINE__, __func__,
605  "gko::Executor (nullptr)");

607  if (!this->is_owning()) {
608  throw gko::NotSupported(__FILE__, __LINE__, __func__,
609  "Non owning gko::Array cannot be resized.");
610  }
611 
612  if (num_elems > 0 && this->is_owning()) {
613  num_elems_ = num_elems;
614  data_.reset(exec_->alloc<value_type>(num_elems));
615  } else {
