613  template <typename T>
614  void copy_from(const Executor *src_exec, size_type num_elems,
615  const T *src_ptr, T *dest_ptr) const
616  {
617  this->template log<log::Logger::copy_started>(
618  src_exec, this, reinterpret_cast<uintptr>(src_ptr),
619  reinterpret_cast<uintptr>(dest_ptr), num_elems * sizeof(T));
620  try {
621  this->raw_copy_from(src_exec, num_elems * sizeof(T), src_ptr,
622  dest_ptr);
623  } catch (NotSupported &) {
624  // Unoptimized copy. Try to go through the masters.
625  auto src_master = src_exec->get_master().get();
626  if (num_elems > 0 && src_master != src_exec) {
627  auto *master_ptr = src_exec->get_master()->alloc<T>(num_elems);
628  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
629  master_ptr);
630  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
631  src_master->free(master_ptr);
632  }
633  }
634  this->template log<log::Logger::copy_completed>(
635  src_exec, this, reinterpret_cast<uintptr>(src_ptr),
636  reinterpret_cast<uintptr>(dest_ptr), num_elems * sizeof(T));
637  }
