805 template <typename ConcreteLoggable, typename PolymorphicBase = Loggable>
806 class EnableLogging : public PolymorphicBase {
807 public:
808  void add_logger(std::shared_ptr<const Logger> logger) override
809  {
810  loggers_.push_back(logger);
811  }
812 
813  void remove_logger(const Logger* logger) override
814  {
815  auto idx =
816  find_if(begin(loggers_), end(loggers_),
817  [&logger](const auto& l) { return l.get() == logger; });
818  if (idx != end(loggers_)) {
819  loggers_.erase(idx);
820  } else {
821  throw OutOfBoundsError(__FILE__, __LINE__, loggers_.size(),
822  loggers_.size());
823  }

847  template <size_type Event, typename ConcreteLoggableT, typename = void>
848  struct propagate_log_helper {
849  template <typename... Args>
850  static void propagate_log(const ConcreteLoggableT*, Args&&...)
851  {}
852  };
853 
854  template <size_type Event, typename ConcreteLoggableT>
855  struct propagate_log_helper<
856  Event, ConcreteLoggableT,
857  xstd::void_t<
858  decltype(std::declval<ConcreteLoggableT>().get_executor())>> {
859  template <typename... Args>
860  static void propagate_log(const ConcreteLoggableT* loggable,
861  Args&&... args)
862  {
863  const auto exec = loggable->get_executor();
864  if (exec->should_propagate_log()) {
865  for (auto& logger : exec->get_loggers()) {
866  if (logger->needs_propagation()) {
867  logger->template on<Event>(std::forward<Args>(args)...);
868  }
869  }
870  }
871  }
872  };
873 
874 protected:
875  template <size_type Event, typename... Params>
876  void log(Params&&... params) const
877  {
878  propagate_log_helper<Event, ConcreteLoggable>::propagate_log(
879  static_cast<const ConcreteLoggable*>(this),
880  std::forward<Params>(params)...);
881  for (auto& logger : loggers_) {
882  logger->template on<Event>(std::forward<Params>(params)...);
