542 template <typename R, typename T>
543 std::unique_ptr<R, std::function<void(R*)>> copy_and_convert_to_impl(
544  std::shared_ptr<const Executor> exec, T* obj)
545 {
546  auto obj_as_r = dynamic_cast<R*>(obj);
547  if (obj_as_r != nullptr && obj->get_executor() == exec) {
548  // FIXME: this breaks lifetimes
549  return {obj_as_r, [](R*) {}};
550  } else {
551  auto copy = R::create(exec);
552  as<ConvertibleTo<std::decay_t<R>>>(obj)->convert_to(copy);
553  return {copy.release(), std::default_delete<R>{}};
554  }
555 }
