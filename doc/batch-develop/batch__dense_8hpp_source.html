398  inline size_type compute_batch_mem(const batch_dim<2>& sizes,
399  const batch_stride& strides)
400  {
401  GKO_ASSERT(sizes.get_num_batch_entries() ==
402  strides.get_num_batch_entries());
403  if (sizes.stores_equal_sizes() && strides.stores_equal_strides()) {
404  return (sizes.at(0))[0] * strides.at(0) *
405  sizes.get_num_batch_entries();
406  }
407  size_type mem_req = 0;
408  for (auto i = 0; i < sizes.get_num_batch_entries(); ++i) {
409  mem_req += (sizes.at(i))[0] * strides.at(i);
410  }
411  return mem_req;
412  }

417  inline batch_stride extract_nth_dim(const int dim, const batch_dim<2>& size)
418  {
419  if (size.stores_equal_sizes()) {
420  return batch_stride(size.get_num_batch_entries(), size.at(0)[dim]);
421  }
422  std::vector<size_type> stride(size.get_num_batch_entries());
423  for (auto i = 0; i < size.get_num_batch_entries(); ++i) {
424  stride[i] = (size.at(i))[dim];
425  }
426  return batch_stride(stride);
427  }

459  inline Array<size_type> compute_num_elems_per_batch_cumul(
460  std::shared_ptr<const Executor> exec, const batch_dim<2>& sizes,
461  const batch_stride& strides)
462  {
463  auto num_elems = Array<size_type>(exec->get_master(),
464  sizes.get_num_batch_entries() + 1);
465  num_elems.get_data()[0] = 0;
466  for (auto i = 0; i < sizes.get_num_batch_entries(); ++i) {
467  num_elems.get_data()[i + 1] =
468  num_elems.get_data()[i] + (sizes.at(i))[0] * strides.at(i);
469  }
470  num_elems.set_executor(exec);
471  return num_elems;
472  }

721 template <typename Matrix, typename... TArgs>
722 std::unique_ptr<Matrix> batch_initialize(
723  std::vector<size_type> stride,
724  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
725  vals,
726  std::shared_ptr<const Executor> exec, TArgs&&... create_args)
727 {
728  using batch_dense = matrix::BatchDense<typename Matrix::value_type>;
729  size_type num_batch_entries = vals.size();
730  std::vector<size_type> num_rows(num_batch_entries);
731  std::vector<dim<2>> sizes(num_batch_entries);
732  auto vals_begin = begin(vals);
733  for (size_type b = 0; b < num_batch_entries; ++b) {
734  num_rows[b] = vals_begin->size();
735  sizes[b] = dim<2>(num_rows[b], 1);
736  vals_begin++;
737  }
738  auto b_size = batch_dim<2>(sizes);
739  auto b_stride = batch_stride(stride);
740  auto tmp = batch_dense::create(exec->get_master(), b_size, b_stride);
741  size_type batch = 0;
742  for (const auto& b : vals) {
743  size_type idx = 0;
744  for (const auto& elem : b) {
745  tmp->at(batch, idx) = elem;
746  ++idx;
747  }
748  ++batch;
749  }
750  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
751  tmp->move_to(mtx.get());
752  return mtx;
753 }

836  size_type cidx = 0;

839  ++cidx;
