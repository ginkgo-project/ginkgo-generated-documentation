482  void validate()
483  {
484  const auto mg_level_len = parameters_.mg_level.size();
485  if (mg_level_len == 0) {
486  GKO_NOT_SUPPORTED(mg_level_len);
487  } else {
488  // each mg_level can not be nullptr
489  for (size_type i = 0; i < mg_level_len; i++) {
490  if (parameters_.mg_level.at(i) == nullptr) {
491  GKO_NOT_SUPPORTED(parameters_.mg_level.at(i));
492  }
493  }
494  }
495  // verify pre-related parameters
496  this->verify_legal_length(true, parameters_.pre_smoother.size(),
497  mg_level_len);
498  // verify post-related parameters when post does not use pre
499  this->verify_legal_length(!parameters_.post_uses_pre,
500  parameters_.post_smoother.size(),
501  mg_level_len);
502  // verify mid-related parameters when mid is standalone smoother.
503  this->verify_legal_length(
504  parameters_.mid_case == multigrid::mid_smooth_type::standalone,
505  parameters_.mid_smoother.size(), mg_level_len);
506  }

517  void verify_legal_length(bool checked, size_type len, size_type ref_len)
518  {
519  if (checked) {
520  // len = 0 uses default behaviour
521  // len = 1 uses the first one
522  // len > 1 : must contain the same len as ref(mg_level)
523  if (len > 1 && len != ref_len) {
524  GKO_NOT_SUPPORTED(this);
525  }
526  }
