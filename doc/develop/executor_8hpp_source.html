698  template <typename T>
699  void copy_from(const Executor* src_exec, size_type num_elems,
700  const T* src_ptr, T* dest_ptr) const
701  {
702  const auto src_loc = reinterpret_cast<uintptr>(src_ptr);
703  const auto dest_loc = reinterpret_cast<uintptr>(dest_ptr);
704  this->template log<log::Logger::copy_started>(
705  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
706  if (this != src_exec) {
707  src_exec->template log<log::Logger::copy_started>(
708  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
709  }
710  try {
711  this->raw_copy_from(src_exec, num_elems * sizeof(T), src_ptr,
712  dest_ptr);
713  } catch (NotSupported&) {
714 #if (GKO_VERBOSE_LEVEL >= 1) && !defined(NDEBUG)
715  // Unoptimized copy. Try to go through the masters.
716  // output to log when verbose >= 1 and debug build
717  std::clog << "Not direct copy. Try to copy data from the masters."
718  << std::endl;
719 #endif
720  auto src_master = src_exec->get_master().get();
721  if (num_elems > 0 && src_master != src_exec) {
722  auto* master_ptr = src_exec->get_master()->alloc<T>(num_elems);
723  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
724  master_ptr);
725  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
726  src_master->free(master_ptr);
727  }
728  }
729  this->template log<log::Logger::copy_completed>(
730  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
731  if (this != src_exec) {
732  src_exec->template log<log::Logger::copy_completed>(
733  src_exec, this, src_loc, dest_loc, num_elems * sizeof(T));
734  }
735  }
