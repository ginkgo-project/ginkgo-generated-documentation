217  void apply_impl(const LinOp* in, LinOp* out) const
218  {
219  auto perm = as<Permutable<index_type>>(in);
220  std::unique_ptr<gko::LinOp> tmp{};
221  if (enabled_permute_ & inverse_permute) {
222  if (enabled_permute_ & row_permute) {
223  tmp = perm->inverse_row_permute(&permutation_);
224  }
225  if (enabled_permute_ & column_permute) {
226  if (enabled_permute_ & row_permute) {
227  tmp = as<Permutable<index_type>>(tmp.get())
228  ->inverse_column_permute(&permutation_);
229  } else {
230  tmp = perm->inverse_column_permute(&permutation_);
231  }
232  }
233  } else {
234  if (enabled_permute_ & row_permute) {
235  tmp = perm->row_permute(&permutation_);
236  }
237  if (enabled_permute_ & column_permute) {
238  if (enabled_permute_ & row_permute) {
239  tmp = as<Permutable<index_type>>(tmp.get())->column_permute(
240  &permutation_);
241  } else {
242  tmp = perm->column_permute(&permutation_);
243  }
244  }
245  }
246  out->copy_from(std::move(tmp));
247  }
