421  void process(const array<index_type>& mtx_row_ptrs,
422  array<index_type>* mtx_srow) override
423  {
424  auto nwarps = mtx_srow->get_num_elems();
425 
426  if (nwarps > 0) {
427  auto host_srow_exec = mtx_srow->get_executor()->get_master();
428  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
429  const bool is_srow_on_host{host_srow_exec ==
430  mtx_srow->get_executor()};
431  const bool is_mtx_on_host{host_mtx_exec ==
432  mtx_row_ptrs.get_executor()};
433  array<index_type> row_ptrs_host(host_mtx_exec);
434  array<index_type> srow_host(host_srow_exec);
435  const index_type* row_ptrs{};
436  index_type* srow{};
437  if (is_srow_on_host) {
438  srow = mtx_srow->get_data();
439  } else {
440  srow_host = *mtx_srow;
441  srow = srow_host.get_data();
442  }
443  if (is_mtx_on_host) {
444  row_ptrs = mtx_row_ptrs.get_const_data();
445  } else {
446  row_ptrs_host = mtx_row_ptrs;
447  row_ptrs = row_ptrs_host.get_const_data();
448  }
449  for (size_type i = 0; i < nwarps; i++) {
450  srow[i] = 0;
451  }
452  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
453  const auto num_elems = row_ptrs[num_rows];
454  const auto bucket_divider =
455  num_elems > 0 ? ceildiv(num_elems, warp_size_) : 1;
456  for (size_type i = 0; i < num_rows; i++) {
457  auto bucket =
458  ceildiv((ceildiv(row_ptrs[i + 1], warp_size_) * nwarps),
459  bucket_divider);
460  if (bucket < nwarps) {
461  srow[bucket]++;
462  }
463  }
464  // find starting row for thread i
465  for (size_type i = 1; i < nwarps; i++) {
466  srow[i] += srow[i - 1];
467  }
468  if (!is_srow_on_host) {
469  *mtx_srow = srow_host;
470  }
471  }
472  }

474  int64_t clac_size(const int64_t nnz) override
475  {
476  if (warp_size_ > 0) {
477  int multiple = 8;
478  if (nnz >= static_cast<int64_t>(2e8)) {
479  multiple = 2048;
480  } else if (nnz >= static_cast<int64_t>(2e7)) {
481  multiple = 512;
482  } else if (nnz >= static_cast<int64_t>(2e6)) {
483  multiple = 128;
484  } else if (nnz >= static_cast<int64_t>(2e5)) {
485  multiple = 32;
486  }
487  if (strategy_name_ == "intel") {
488  multiple = 8;
489  if (nnz >= static_cast<int64_t>(2e8)) {
490  multiple = 256;
491  } else if (nnz >= static_cast<int64_t>(2e7)) {
492  multiple = 32;
493  }
494  }
495 #if GINKGO_HIP_PLATFORM_HCC
496  if (!cuda_strategy_) {
497  multiple = 8;
498  if (nnz >= static_cast<int64_t>(1e7)) {
499  multiple = 64;
500  } else if (nnz >= static_cast<int64_t>(1e6)) {
501  multiple = 16;
502  }
503  }
504 #endif // GINKGO_HIP_PLATFORM_HCC
505 
506  auto nwarps = nwarps_ * multiple;
507  return min(ceildiv(nnz, warp_size_), nwarps);
508  } else {
509  return 0;
510  }
511  }

612  void process(const array<index_type>& mtx_row_ptrs,
613  array<index_type>* mtx_srow) override
614  {
615  // if the number of stored elements is larger than <nnz_limit> or
616  // the maximum number of stored elements per row is larger than
617  // <row_len_limit>, use load_balance otherwise use classical
618  index_type nnz_limit = nvidia_nnz_limit;
619  index_type row_len_limit = nvidia_row_len_limit;
620  if (strategy_name_ == "intel") {
621  nnz_limit = intel_nnz_limit;
622  row_len_limit = intel_row_len_limit;
623  }
624 #if GINKGO_HIP_PLATFORM_HCC
625  if (!cuda_strategy_) {
626  nnz_limit = amd_nnz_limit;
627  row_len_limit = amd_row_len_limit;
628  }
629 #endif // GINKGO_HIP_PLATFORM_HCC
630  auto host_mtx_exec = mtx_row_ptrs.get_executor()->get_master();
631  const bool is_mtx_on_host{host_mtx_exec ==
632  mtx_row_ptrs.get_executor()};
633  array<index_type> row_ptrs_host(host_mtx_exec);
634  const index_type* row_ptrs{};
635  if (is_mtx_on_host) {
636  row_ptrs = mtx_row_ptrs.get_const_data();
637  } else {
638  row_ptrs_host = mtx_row_ptrs;
639  row_ptrs = row_ptrs_host.get_const_data();
640  }
641  const auto num_rows = mtx_row_ptrs.get_num_elems() - 1;
642  if (row_ptrs[num_rows] > nnz_limit) {
643  load_balance actual_strategy(nwarps_, warp_size_,
644  cuda_strategy_, strategy_name_);
645  if (is_mtx_on_host) {
646  actual_strategy.process(mtx_row_ptrs, mtx_srow);
647  } else {
648  actual_strategy.process(row_ptrs_host, mtx_srow);
649  }
650  this->set_name(actual_strategy.get_name());
651  } else {
652  index_type maxnum = 0;
653  for (size_type i = 0; i < num_rows; i++) {
654  maxnum = std::max(maxnum, row_ptrs[i + 1] - row_ptrs[i]);
655  }
656  if (maxnum > row_len_limit) {
657  load_balance actual_strategy(
658  nwarps_, warp_size_, cuda_strategy_, strategy_name_);
659  if (is_mtx_on_host) {
660  actual_strategy.process(mtx_row_ptrs, mtx_srow);
661  } else {
662  actual_strategy.process(row_ptrs_host, mtx_srow);
663  }
664  this->set_name(actual_strategy.get_name());
665  } else {
666  classical actual_strategy;
667  if (is_mtx_on_host) {
668  actual_strategy.process(mtx_row_ptrs, mtx_srow);
669  max_length_per_row_ =
670  actual_strategy.get_max_length_per_row();
671  } else {
672  actual_strategy.process(row_ptrs_host, mtx_srow);
673  max_length_per_row_ =
674  actual_strategy.get_max_length_per_row();
675  }
676  this->set_name(actual_strategy.get_name());
677  }
678  }
679  }
