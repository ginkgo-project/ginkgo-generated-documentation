279  explicit Ic(const Factory* factory, std::shared_ptr<const LinOp> lin_op)
280  : EnableLinOp<Ic>(factory->get_executor(), lin_op->get_size()),
281  parameters_{factory->get_parameters()}
282  {
283  auto comp =
284  std::dynamic_pointer_cast<const Composition<value_type>>(lin_op);
285  std::shared_ptr<const LinOp> l_factor;
286 
287  // build factorization if we weren't passed a composition
288  if (!comp) {
289  auto exec = lin_op->get_executor();
290  if (!parameters_.factorization_factory) {
291  parameters_.factorization_factory =
292  factorization::ParIc<value_type, index_type>::build()
293  .with_both_factors(false)
294  .on(exec);
295  }
296  auto fact = std::shared_ptr<const LinOp>(
297  parameters_.factorization_factory->generate(lin_op));
298  // ensure that the result is a composition
299  comp =
300  std::dynamic_pointer_cast<const Composition<value_type>>(fact);
301  if (!comp) {
302  GKO_NOT_SUPPORTED(comp);
303  }
304  }
305  // comp must contain one or two factors
306  if (comp->get_operators().size() > 2 || comp->get_operators().empty()) {
307  GKO_NOT_SUPPORTED(comp);
308  }
309  l_factor = comp->get_operators()[0];
310  GKO_ASSERT_IS_SQUARE_MATRIX(l_factor);
311 
312  auto exec = this->get_executor();
313 
314  // If no factories are provided, generate default ones
315  if (!parameters_.l_solver_factory) {
316  l_solver_ = generate_default_solver<l_solver_type>(exec, l_factor);
317  // If comp contains both factors: use the transposed factor to avoid
318  // transposing twice
319  if (comp->get_operators().size() == 2) {
320  auto lh_factor = comp->get_operators()[1];
321  GKO_ASSERT_EQUAL_DIMENSIONS(l_factor, lh_factor);
322  lh_solver_ = as<lh_solver_type>(l_solver_->conj_transpose());
323  } else {
324  lh_solver_ = as<lh_solver_type>(l_solver_->conj_transpose());
325  }
326  } else {
327  l_solver_ = parameters_.l_solver_factory->generate(l_factor);
328  lh_solver_ = as<lh_solver_type>(l_solver_->conj_transpose());
329  }
330  }

361  const std::shared_ptr<const LinOp>& mtx)
362  {

369  .with_max_iters(default_max_iters)
