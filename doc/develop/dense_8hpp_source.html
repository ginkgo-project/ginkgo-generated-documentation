1200 template <typename Matrix, typename... TArgs>
1201 std::unique_ptr<Matrix> initialize(
1202  size_type stride, std::initializer_list<typename Matrix::value_type> vals,
1203  std::shared_ptr<const Executor> exec, TArgs&&... create_args)
1204 {
1205  using dense = matrix::Dense<typename Matrix::value_type>;
1206  size_type num_rows = vals.size();
1207  auto tmp = dense::create(exec->get_master(), dim<2>{num_rows, 1}, stride);
1208  size_type idx = 0;
1209  for (const auto& elem : vals) {
1210  tmp->at(idx) = elem;
