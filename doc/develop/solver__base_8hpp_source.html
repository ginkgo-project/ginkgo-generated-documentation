554 public:

556 
557  EnablePreconditionedIterativeSolver(
558  std::shared_ptr<const LinOp> system_matrix,
559  std::shared_ptr<const stop::CriterionFactory> stop_factory,
560  std::shared_ptr<const LinOp> preconditioner)
561  : EnableSolverBase<DerivedType>(std::move(system_matrix)),
562  EnableIterativeBase<DerivedType>{std::move(stop_factory)},

565 
566  template <typename FactoryParameters>
567  EnablePreconditionedIterativeSolver(
568  std::shared_ptr<const LinOp> system_matrix,
569  const FactoryParameters& params)
570  : EnablePreconditionedIterativeSolver{
571  system_matrix, stop::combine(params.criteria),
572  generate_preconditioner(system_matrix, params)}
573  {}
574 
575 private:
576  template <typename FactoryParameters>
577  static std::shared_ptr<const LinOp> generate_preconditioner(
578  std::shared_ptr<const LinOp> system_matrix,
579  const FactoryParameters& params)
580  {
581  if (params.generated_preconditioner) {
582  return params.generated_preconditioner;
583  } else if (params.preconditioner) {
584  return params.preconditioner->generate(system_matrix);
585  } else {
586  return matrix::Identity<ValueType>::create(
587  system_matrix->get_executor(), system_matrix->get_size());
588  }
589  }
