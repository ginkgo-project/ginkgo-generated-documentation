486 template <typename Matrix, typename... TArgs>
487 std::unique_ptr<Matrix> initialize(
488  std::initializer_list<std::initializer_list<typename Matrix::value_type>>
489  vals,
490  std::shared_ptr<const Executor> exec, TArgs&&... create_args)
491 {
492  using batch_multi_vector = MultiVector<typename Matrix::value_type>;
493  size_type num_batch_items = vals.size();
494  GKO_THROW_IF_INVALID(num_batch_items > 0, "Input data is empty");
495  auto vals_begin = begin(vals);
496  size_type common_num_rows = vals_begin ? vals_begin->size() : 0;
497  auto common_size = dim<2>(common_num_rows, 1);
498  for (auto& val : vals) {
499  GKO_ASSERT_EQ(common_num_rows, val.size());
500  }
501  auto b_size = batch_dim<2>(num_batch_items, common_size);
502  auto tmp = batch_multi_vector::create(exec->get_master(), b_size);
503  size_type batch = 0;
504  for (const auto& b : vals) {
505  size_type idx = 0;
506  for (const auto& elem : b) {
507  tmp->at(batch, idx) = elem;
508  ++idx;
509  }
510  ++batch;
511  }
512  auto mtx = Matrix::create(exec, std::forward<TArgs>(create_args)...);
513  tmp->move_to(mtx);
514  return mtx;
515 }
