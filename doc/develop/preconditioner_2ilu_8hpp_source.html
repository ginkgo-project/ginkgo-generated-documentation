267  {
268  // take care of real-to-complex apply
269  precision_dispatch_real_complex<value_type>(
270  [&](auto dense_b, auto dense_x) {
271  this->set_cache_to(dense_b);
272  if (!ReverseApply) {
273  l_solver_->apply(dense_b, cache_.intermediate.get());
274  if (u_solver_->apply_uses_initial_guess()) {
275  dense_x->copy_from(cache_.intermediate.get());
276  }
277  u_solver_->apply(cache_.intermediate.get(), dense_x);
278  } else {
279  u_solver_->apply(dense_b, cache_.intermediate.get());

312  explicit Ilu(const Factory* factory, std::shared_ptr<const LinOp> lin_op)
313  : EnableLinOp<Ilu>(factory->get_executor(), lin_op->get_size()),
314  parameters_{factory->get_parameters()}
315  {
316  auto comp =
317  std::dynamic_pointer_cast<const Composition<value_type>>(lin_op);
318  std::shared_ptr<const LinOp> l_factor;
319  std::shared_ptr<const LinOp> u_factor;
320 
321  // build factorization if we weren't passed a composition
322  if (!comp) {
323  auto exec = lin_op->get_executor();
324  if (!parameters_.factorization_factory) {
325  parameters_.factorization_factory =
326  factorization::ParIlu<value_type, index_type>::build().on(
327  exec);
328  }
329  auto fact = std::shared_ptr<const LinOp>(
330  parameters_.factorization_factory->generate(lin_op));
331  // ensure that the result is a composition
332  comp =
333  std::dynamic_pointer_cast<const Composition<value_type>>(fact);
334  if (!comp) {
335  GKO_NOT_SUPPORTED(comp);
336  }
337  }
338  if (comp->get_operators().size() == 2) {
339  l_factor = comp->get_operators()[0];
340  u_factor = comp->get_operators()[1];
341  } else {
342  GKO_NOT_SUPPORTED(comp);
343  }
344  GKO_ASSERT_EQUAL_DIMENSIONS(l_factor, u_factor);
345 
346  auto exec = this->get_executor();
347 
348  // If no factories are provided, generate default ones
349  if (!parameters_.l_solver_factory) {
350  l_solver_ = generate_default_solver<l_solver_type>(exec, l_factor);
351  } else {
352  l_solver_ = parameters_.l_solver_factory->generate(l_factor);
353  }
354  if (!parameters_.u_solver_factory) {
355  u_solver_ = generate_default_solver<u_solver_type>(exec, u_factor);
356  } else {
357  u_solver_ = parameters_.u_solver_factory->generate(u_factor);
358  }
359  }

386  template <typename SolverType>
387  static std::enable_if_t<solver::has_with_criteria<SolverType>::value,
388  std::unique_ptr<SolverType>>
389  generate_default_solver(const std::shared_ptr<const Executor>& exec,
390  const std::shared_ptr<const LinOp>& mtx)
391  {
392  constexpr gko::remove_complex<value_type> default_reduce_residual{1e-4};

395 
396  return SolverType::build()
397  .with_criteria(gko::stop::Iteration::build()
398  .with_max_iters(default_max_iters)

402  .on(exec))
