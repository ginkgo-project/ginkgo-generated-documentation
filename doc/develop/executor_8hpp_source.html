782  template <typename T>
783  void copy_from(ptr_param<const Executor> src_exec, size_type num_elems,
784  const T* src_ptr, T* dest_ptr) const
785  {
786  const auto src_loc = reinterpret_cast<uintptr>(src_ptr);
787  const auto dest_loc = reinterpret_cast<uintptr>(dest_ptr);
788  this->template log<log::Logger::copy_started>(
789  src_exec.get(), this, src_loc, dest_loc, num_elems * sizeof(T));
790  if (this != src_exec.get()) {
791  src_exec->template log<log::Logger::copy_started>(
792  src_exec.get(), this, src_loc, dest_loc, num_elems * sizeof(T));
793  }
794  try {
795  this->raw_copy_from(src_exec.get(), num_elems * sizeof(T), src_ptr,
796  dest_ptr);
797  } catch (NotSupported&) {
798 #if (GKO_VERBOSE_LEVEL >= 1) && !defined(NDEBUG)
799  // Unoptimized copy. Try to go through the masters.
800  // output to log when verbose >= 1 and debug build
801  std::clog << "Not direct copy. Try to copy data from the masters."
802  << std::endl;
803 #endif
804  auto src_master = src_exec->get_master().get();
805  if (num_elems > 0 && src_master != src_exec.get()) {
806  auto* master_ptr = src_exec->get_master()->alloc<T>(num_elems);
807  src_master->copy_from<T>(src_exec, num_elems, src_ptr,
808  master_ptr);
809  this->copy_from<T>(src_master, num_elems, master_ptr, dest_ptr);
810  src_master->free(master_ptr);
811  }
812  }
813  this->template log<log::Logger::copy_completed>(
814  src_exec.get(), this, src_loc, dest_loc, num_elems * sizeof(T));
815  if (this != src_exec.get()) {
816  src_exec->template log<log::Logger::copy_completed>(
817  src_exec.get(), this, src_loc, dest_loc, num_elems * sizeof(T));
818  }
819  }
